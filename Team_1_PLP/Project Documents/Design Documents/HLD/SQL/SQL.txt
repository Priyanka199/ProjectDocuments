	CREATE TABLE USERS 
   (USER_ID NUMBER(4,0), 
	USER_NAME VARCHAR2(20 BYTE), 
	PASSWRD VARCHAR2(20 BYTE), 
	EMAIL VARCHAR2(30 BYTE), 
	LIBRARIAN NUMBER(10,0), 
	PRIMARY KEY (USER_ID));
	 
	CREATE TABLE BOOK_INVENTORY
   (BOOK_ID NUMBER(4,0), 
	BOOK_NAME VARCHAR2(20 BYTE), 
	AUTHOR1 VARCHAR2(20 BYTE), 
	AUTHOR2 VARCHAR2(20 BYTE), 
	PUBLISHER VARCHAR2(20 BYTE), 
	YEAROFPUBLICATION VARCHAR2(4 BYTE));
	
	CREATE TABLE BOOK_TRANSACTION 
   (TRANSACTION_ID NUMBER(4,0), 
	REG_ID NUMBER(4,0), 
	ISSUE_DATE DATE, 
	RETURN_DATE DATE, 
	FINE NUMBER(4,0), 
	STUD_RETURN_DATE DATE);
	
	CREATE TABLE BOOKSREGISTRATION
   (REGISTRATION_ID VARCHAR2(4 BYTE), 
	BOOK_ID VARCHAR2(4 BYTE), 
	USER_ID VARCHAR2(4 BYTE), 
	REGISTRATIONDATE DATE, 
	GRANT_STATUS NUMBER(2,0), 
	PRIMARY KEY (REGISTRATION_ID));
